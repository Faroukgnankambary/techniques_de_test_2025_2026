@app.route("/triangulate", methods=["POST"])
def triangulate() -> Response:
    """
    Endpoint principal.

    CAS ACCEPTÉS :
    1. JSON : {"pointset_id": "..."}
       → Récupère le pointset auprès du PointSetManager

    2. Corps binaire direct
       → Interprété comme un PointSet binaire

    Retour :
    - Binaire des triangles si succès
    - JSON d'erreur si problème
    """

    # ------------------------------------------------------------------
    # CAS 1 : JSON avec pointset_id
    # ------------------------------------------------------------------
    if request.is_json:
        payload = request.get_json(silent=True)
        if not payload or "pointset_id" not in payload:
            return jsonify({"error": "JSON invalide, 'pointset_id' manquant"}), 400

        pointset_id = payload["pointset_id"]

        try:
            data = pointset_client.fetch_pointset(pointset_id)
        except Exception as e:
            return jsonify({"error": str(e)}), 503

        try:
            points = binary_to_pointset(data)
        except Exception:
            return jsonify({"error": "PointSet invalide"}), 400

    # ------------------------------------------------------------------
    # CAS 2 : Données binaires envoyées directement
    # ------------------------------------------------------------------
    else:
        data = request.get_data()

        if not data:
            return jsonify({"error": "Aucune donnée fournie"}), 400

        try:
            points = binary_to_pointset(data)
        except Exception:
            return jsonify({"error": "Format binaire non valide"}), 400

    # ------------------------------------------------------------------
    # TRIANGULATION
    # ------------------------------------------------------------------
    try:
        triangles = simple_triangulation(points)
    except Exception:
        return jsonify({"error": "Échec de la triangulation"}), 500

    # ------------------------------------------------------------------
    # SÉRIALISATION
    # ------------------------------------------------------------------
    try:
        binary_output = triangles_to_binary(points, triangles)
    except Exception:
        return jsonify({"error": "Échec sérialisation triangles"}), 500

    # Réponse binaire brute
    return Response(binary_output, status=200, mimetype="application/octet-stream")

class PointSetManagerClient:
    """
    Client minimaliste du PointSetManager.
    Implémenté UNIQUEMENT avec la bibliothèque standard (urllib)
    car requests est interdit dans le sujet.
    """

    def __init__(self, base_url: str = "http://localhost:8000"):
        self.base_url = base_url.rstrip("/")

    def fetch_pointset(self, pointset_id: str) -> bytes:
        """
        Récupère un PointSet en binaire depuis le PointSetManager.
        """
        url = f"{self.base_url}/pointset/{pointset_id}"

        try:
            with urllib.request.urlopen(url) as response:
                return response.read()

        except urllib.error.HTTPError as e:
            raise RuntimeError(f"Erreur HTTP du PointSetManager : {e.code}") from e

        except urllib.error.URLError as e:
            raise RuntimeError("Impossible de contacter PointSetManager") from e

    def health_check(self) -> bool:
        """
        Vérifie si le PointSetManager répond.
        """
        url = f"{self.base_url}/health"

        try:
            with urllib.request.urlopen(url) as response:
                data = response.read()
                return b"ok" in data.lower()
        except Exception:
            return False
